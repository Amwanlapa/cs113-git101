week 6 Array

อาเรย์ คือ กลุ่มของตัวเเปรที่ถูกจัดการร่วมกัน ตัวเเปรต้องเป็นประเภทเดียวกัน
1. การประกาศอาเรย์
  int n[5]; จะเป็นการประกาศตัวแปรประเภท integer มา 5 ตัว โดยในอาเรย์เราจะใช้ index 
  ในการบอกตำแหน่ง เช่น n[0], n[1] ไปจนถึง n[4] นั่นคือ index ของอาเรย์จะเริ่มจาก 0 ถึง size-1 เสมอ
2. การอ่านค่าในอาเรย์
ex.
#include <stdio.h>
int main()
{
    int n[8] = {5, 10, 15, 20, 25, 30, 35, 40};

    printf("n[2] = %d\n", n[2]);
    printf("n[3] = %d\n", n[3]);

    n[2] = 99;
    printf("n[2] = %d\n", n[2]);
    return 0;
}
output  
n[2] = 15
n[3] = 20
n[2] = 99
3.การใช้คำสั่ง for loop กับอาเรย์
ex.
#include <stdio.h>
int main()
{
    int n[8] = {5, 10, 15, 20, 25, 30, 35, 40};

    int i;
    for (i = 0; i < 8; i++)
    {
        printf("%d, ", n[i]);
    }

    printf("\n");

    for (i = 7; i >= 0; i--)
    {
        printf("%d, ", n[i]);
    }
    return 0;
}
อันแรกเป็นการอ่านค่าของอาเรย์แบบปกติและแบบที่สองอ่านจากตำแหน่งสุดท้ายมายังตำแหน่งแรก
output
5, 10, 15, 20, 25, 30, 35, 40,
40, 35, 30, 25, 20, 15, 10, 5,
4.อาเรย์ 2 มิติ
  int A[2][3];
  [2] = row ความสูง
  [3] = column ความยาว
ex.
int a[3][4] = {
    {35, 6, 1},
    {4, 5, 16},
    {9, 7, 92}
};

char b[5][10];
 ถ้าต้องการเลข 7 จะอยู่ที่ตำเเหน่ง a[2][1]
